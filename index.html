Implement a solution to the Wolves and Chicks Puzzle from the game Professor Layton and the Curious Village.

Get the three wolves and three chicks seen below to the other side of the river while obeying the following conditions.

No more than two animals can ride the raft at the same time.

There must be at least one animal on the raft in order for it to move.

If more wolves than chicks stay on either side of the river, the wolves will eat the chicks, and you'll have to start over.

You can move the raft as many times as you like, but this feat can be accomplished in as few as 11 moves.

Implementation of Depth First Search

We will now implement the depth first search algorithm to solve the puzzle.

I will use an implementation using a queue instead of a recursive solution for reasons that will become apparent later

def search( initial, successorFunc, limit = 1000 ):
    queue = [ ( initial, [initial] ) ]
    numNodes = 1
    sol = None
    while( len( queue ) > 0 ):
        # current, chist = queue[-1]
        # queue = queue[0:-1]
        current, chist = queue.pop()
        if (current[1] == 0) and (current[2] == 0):
            sol = (numNodes, chist)
            break
        #print("current state", current, chist )
        #print("queue", queue)
        children = successorFunc( current )
        #print("children", children)
        for c in children:
            if c not in chist:
                queue.append((c, chist + [c] ))
                numNodes = numNodes + 1
                if ( numNodes > limit ):
                    return None
    return sol

NUM_ANIMALS = 3


def is_valid(state):
  wolves_left, chicks_left, boat_left = state
  wolves_right, chicks_right = NUM_ANIMALS - wolves_left, NUM_ANIMALS - chicks_left
  boat_right = not boat_left

  #print("is_valid", wolves_left, chicks_left, boat_left, wolves_right, chicks_right, boat_right)

  if ((wolves_left < 0) or (wolves_left > NUM_ANIMALS)):
    return False
  if ((chicks_left < 0) or (chicks_left > NUM_ANIMALS)):
    return False
  if ((chicks_left > 0) and
      (wolves_left > chicks_left)) or ((chicks_right > 0) and
                                       (wolves_right > chicks_right)):
    return False

  return True

def expand_node(state):
  wolves_left, chicks_left, boat_left = state
  wolves_right, chicks_right = NUM_ANIMALS - wolves_left, NUM_ANIMALS - chicks_left
  boat_right = not boat_left
  move_left = [[0, 1], [1, 0], [1, 1], [0, 2], [2, 0]]
  move_right = [[0, -1], [-1, 0], [-1, -1], [0, -2], [-2, 0]]

  children = []

  move = move_left if boat_left else move_right

  for mw, mc in move:
    #print('move', mw, mc)
    n_wolves_left = wolves_left - mw
    n_chicks_left = chicks_left - mc
    child = [n_wolves_left, n_chicks_left, not boat_left]
    if is_valid(child):
      children.append(child)
  return children

NODE_LIMIT = 2000

def is_goal(state):
  res = False
  if ((state[0] == 0) and (state[1] == 0)):
    res = True
  return res

def search(dfs=True):
  initial_state = [NUM_ANIMALS, NUM_ANIMALS, True]
  queue = [[initial_state, []]]
  count = 0
  max_len_queue = 0

  sol = None

  while (len(queue) > 0):
    if (len(queue) > max_len_queue):
      max_len_queue = len(queue)
    current_state, current_hist = queue.pop()
    #print('current', current_state, current_hist)
    if (is_goal(current_state)):
      sol = current_hist + [current_state]
    children = expand_node(current_state)
    for child in children:
      if (child not in current_hist):
        if (dfs):
          ins = len(queue)
        else:
          ins = 0
        queue.insert(ins, [child, current_hist + [current_state]])

    count = count + 1
    if (count > NODE_LIMIT):
      print("Too many nodes")
      break

  return sol, count, max_len_queue


sol, count, memory = search(dfs=True)
